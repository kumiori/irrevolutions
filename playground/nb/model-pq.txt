\begin{equation}\label{ eqn:pq-a }a(\alpha)=\left(1 - \alpha{\left(x \right)}\right)^{q} \left(\alpha{\left(x \right)} + 1\right)^{- p}\end{equation}
\begin{equation}\label{ eqn:pq-aprime }a'(\alpha)=- \frac{p \left(1 - \alpha{\left(x \right)}\right)^{q} \left(\alpha{\left(x \right)} + 1\right)^{- p}}{\alpha{\left(x \right)} + 1} - \frac{q \left(1 - \alpha{\left(x \right)}\right)^{q} \left(\alpha{\left(x \right)} + 1\right)^{- p}}{1 - \alpha{\left(x \right)}}\end{equation}
\begin{equation}\label{ eqn:pq-apprime }a''(\alpha)=\left(1 - \alpha{\left(x \right)}\right)^{q} \left(\alpha{\left(x \right)} + 1\right)^{- p} \left(- \frac{2 p q}{\left(\alpha{\left(x \right)} - 1\right) \left(\alpha{\left(x \right)} + 1\right)} + \frac{p \left(p + 1\right)}{\left(\alpha{\left(x \right)} + 1\right)^{2}} + \frac{q \left(q - 1\right)}{\left(\alpha{\left(x \right)} - 1\right)^{2}}\right)\end{equation}
\begin{equation}\label{ eqn:pq-s }s(\alpha)=\left(1 - \alpha{\left(x \right)}\right)^{- q} \left(\alpha{\left(x \right)} + 1\right)^{p}\end{equation}
\begin{equation}\label{ eqn:pq-sprime }s'(\alpha)=\frac{p \left(1 - \alpha{\left(x \right)}\right)^{- q} \left(\alpha{\left(x \right)} + 1\right)^{p}}{\alpha{\left(x \right)} + 1} + \frac{q \left(1 - \alpha{\left(x \right)}\right)^{- q} \left(\alpha{\left(x \right)} + 1\right)^{p}}{1 - \alpha{\left(x \right)}}\end{equation}
\begin{equation}\label{ eqn:pq-spprime }s''(\alpha)=\left(1 - \alpha{\left(x \right)}\right)^{- q} \left(\alpha{\left(x \right)} + 1\right)^{p} \left(- \frac{2 p q}{\left(\alpha{\left(x \right)} - 1\right) \left(\alpha{\left(x \right)} + 1\right)} + \frac{p \left(p - 1\right)}{\left(\alpha{\left(x \right)} + 1\right)^{2}} + \frac{q \left(q + 1\right)}{\left(\alpha{\left(x \right)} - 1\right)^{2}}\right)\end{equation}
